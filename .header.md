# How to use this module




```
module "vault_dynamic_secret_mysql" {
  source = "../.."

  vault_mount_path = "database"
  db_username = "vault-user"
  db_password = "password"
  db_url = "mysql:3306"

  roles = [
    {
      role_name: "main"
      database_name: "wp-vault-test"
    },
    {
      role_name: "dnb"
      database_name: "wp-1"
    },
  ]
}
```
